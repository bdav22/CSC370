import java.util.Arrays;

public class Dirsort {

	public static String[] sort(String[] dirs) {

		// variable intializations
		int[] numSlashes = {};
		int mostSlashes = 0;
		int a = 0;
		String[] sorted = {};
		String temp;

		
		// loop through string and have a "slash" counter
		for (int i = 0; i < dirs.length; i++) {

			
			// reset numslashes to 0 for each element of dirs
			numSlashes[i] = 0;

			// loop through a specific element in dirs
			for (int j = 0; j < dirs[i].length(); j++) {

				// if there is a slash
				if (dirs[i].charAt(j) == '/') {

					// increment numslashes at i
					numSlashes[i]++;

				}

				// increase the value of mostslashes if needed
				if (numSlashes[i] >= mostSlashes) {

					mostSlashes = numSlashes[i];

				}

			}

		}

		// loop through mostslashes and sort based on number of slashes
		for (int i = 0; i < mostSlashes; i++) {

			for (int j = 0; j < dirs.length; j++) {

				if (numSlashes[j] == i) {

					// change sorted at a
					sorted[a] = dirs[j];

					// increment a
					a++;

				}

			}

		}

		// number of slashes are re-found after sorting; need to do this to now sort by alphabet
		for (int i = 0; i < sorted.length; i++) {

			numSlashes[i] = 0;

			// loop through the individual element in the sorted array
			for (int j = 0; j < sorted[i].length(); j++) {

				// if there is a slash
				if (sorted[i].charAt(j) == '/') {

					// increment numslashes at i
					numSlashes[i]++;

				}

			}

		}

		// loop through most slashes
		for (int i = 0; i < mostSlashes; i++) {

			// loop through the sorted array
			for (int j = 0; j < (sorted.length - 1); j++) {

				if (numSlashes[j] == i && numSlashes[j + 1] == j) {

					// sorting algorithm
					if (sorted[j].compareTo(sorted[j + 1]) > 0) {

						// swap
						temp = sorted[j];
						sorted[j] = sorted[j + 1];
						sorted[j + 1] = temp;

						j = 0;

					} // most inside if

				}

			}

		}

		return sorted;

	}
